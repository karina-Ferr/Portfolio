# PACOTES
# ------------------------------------------------------
library(dismo)     # modelagem de nicho
library(raster)    # dados espaciais
library(sp)        # manipula√ß√£o espacial
library(rgdal)     # leitura de shapefiles

# ------------------------------------------------------
# DADOS SIMULADOS
# ------------------------------------------------------

# ocorr√™ncias de esp√©cies (exemplo)
ocorrencias <- data.frame(
  lon = runif(20, -50, -45),
  lat = runif(20, -23, -20)
)
coordinates(ocorrencias) <- ~ lon + lat

# vari√°veis ambientais (raster simulado)
temp <- raster(matrix(runif(100, 15, 30), nrow=10))
prec <- raster(matrix(runif(100, 800, 2000), nrow=10))

# ------------------------------------------------------
# MODELAGEM DE NICHO (exemplo simples)
# ------------------------------------------------------

# MaxEnt ou GLM ‚Äî aqui s√≥ exemplo

# ------------------------------------------------------
# MAPA DE PROBABILIDADE
# ------------------------------------------------------

# ------------------------------------------------------
# SOBREPOSI√á√ÉO COM √ÅREAS DE RISCO (simulado)
# ------------------------------------------------------

# √°rea de risco (shapefile fict√≠cio)

# simula√ß√£o:
risco <- as(extent(-49, -46, -22.5, -21), "SpatialPolygons")

plot(mapa_prob)
plot(risco, add=TRUE, border="red", lwd=2)
